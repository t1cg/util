class Log {
    constructor(discard, level, stream) {
        this.level = level;
        if (discard) {
            // don't write to file
            this.out = new DiscardStream();
        }
        else {
            // write to file
            this.out = stream;
        }
        this.prefix = moment().format() + " " + level + " | ";
    }
    Output(msg) {
        this.out.write(this.prefix + msg + "\n");
    }
}
